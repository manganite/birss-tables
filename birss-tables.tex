\section{Content Overview}

This document compiles Tables~3, 4a--4f, 6, and~7 from R.\,R.\ Birss,
\textit{Symmetry and Magnetism} (Elsevier/\allowbreak North-Holland, 1966),
typeset with proper crystallographic notation.

\begin{notelist}
\item \textbf{Table~3} (p.\,\pageref{sec:table3}): The 32 classical crystallographic
  point groups --- Hermann--Mauguin (international) symbols, Schoenflies and Shubnikov
  symbols, symmetry operations, group order, and generating matrices.
\item \textbf{Table~4a} (p.\,\pageref{sec:table4a}): Maps each of the 32 point groups to
  tensor-symbol classes (letters A--U) for four tensor types (polar/axial $\times$
  even/odd rank).
\item \textbf{Tables~4b--4f} (p.\,\pageref{sec:table4bcd}): Independent tensor components
  for each symbol class at ranks 0--4. Table~4b: rank~0 (scalar); Table~4c: rank~1
  (vector); Table~4d: rank~2; Table~4e: rank~3; Table~4f: rank~4.
\item \textbf{Table~6} (p.\,\pageref{sec:table6}): The 90 magnetic point groups (types I and III)
  (90 rows: 32 classical + 58 black-and-white; the 32 grey groups are omitted).
\item \textbf{Table~7} (p.\,\pageref{sec:table7}): i-tensor and c-tensor symbol classes
  for the 90 magnetic point groups, using the same A--U scheme as Table~4a.
\end{notelist}

\section{Notation}

\subsection*{Point-group symbols}
\begin{notelist}
\item \textbf{Overbars} denote improper rotations (roto-inversions):
  $\ob{1}$ = inversion, $\ob{2}$ = mirror reflection,
  $\ob{3}$, $\ob{4}$, $\ob{6}$ = higher-order roto-inversions.
\item \textbf{Hermann--Mauguin} (international) symbols: e.g.\ $\ob{4}2m$,
  $m\ob{3}m$, $6/mmm$.
\item \textbf{Schoenflies} symbols: e.g.\ $C_{2v}$, $D_{4h}$, $O_h$.
\item \textbf{Shubnikov} symbols: use colons ($:$) for normal subgroups, dots
  ($\cdot$) for semidirect products, slashes ($/$) for quotients. Overbars as above.
\end{notelist}

\subsection*{Symmetry operations}
\begin{notelist}
\item $\pm n_z$ = the pair of proper $n$-fold rotations about $z$.
\item $\pm\ob{n}_z$ = the pair of $n$-fold roto-inversions about $z$.
\item $\perp$ = perpendicular to the principal axis (e.g.\ $3(2{\perp})$ =
  three 2-fold axes perpendicular to the 3-fold axis).
\item $n(m)$ groups $n$ equivalent operations of type $m$.
\item $2_{xy}$, $2_{-xy}$ = diagonal 2-fold axes (along $[110]$- and
  $[1\ob{1}0]$-type directions); similarly $\ob{2}_{xy}$, $\ob{2}_{-xy}$ for
  the corresponding mirrors.
\item A prime ($'$) indicates the operation is combined with time reversal
  (Tables~6 and~7). In grouped terms $n(m'{\perp})$, the prime is on the
  operation symbol, not the multiplicity count.
\end{notelist}

\subsection*{Generating matrices}
\begin{notelist}
\item $\sg{N}$: named generating matrices (see Birss \S5 for definitions).
\item $\sgp{N}$: generating matrix combined with time reversal (Table~6).
\end{notelist}

\subsection*{Tensor symbol classes and components (Tables 4a--4f, 7)}
\begin{notelist}
\item Letters A--U label 21 distinct symmetry classes. Table~4a (and Table~7)
  assigns each point group a letter for each tensor type; this letter is the
  row key into Tables~4b--4f.
\item A dash (---) in Tables~4a or~7 means the tensor vanishes identically.
\item In Tables~4b--4f, column headers are tensor index tuples.
  Permutation shorthands: $(3)$ = 3 unrestricted permutations;
  $(x{\cdot}3)$ = 3 permutations keeping a specific index structure;
  $(x{:}3)$ = 3 permutations with the first index fixed;
  $(c4)$ = 4 cyclic permutations;
  $(xy{:}6)$ = 6 permutations preserving the relative order of $x$ and $y$.
\item A coordinate symbol (e.g.\ $xx$, $xyz$) means the component survives as an
  independent parameter. A minus sign (e.g.\ $-xy$) means the component equals
  the negative of the named one. A compound entry (e.g.\ $yyxx{+}xyyx{+}yxyx$)
  means the component equals that sum. $0$ means the component vanishes.
\item Parenthesized tensor symbols such as $(E_m)$ in Table~7 denote alternate
  reference-axis settings (same abstract group, rotated axes).
\end{notelist}

\section{How to Use These Tables}

\subsection*{Finding tensor forms for a classical point group}
\begin{enumerate}\setlength{\itemsep}{1pt}
\item Find the point group in \textbf{Table~3} to see its symmetry operations and
  generating matrices.
\item In \textbf{Table~4a}, read the symbol class (A--U) for the desired tensor type
  (polar/axial, even/odd rank).
\item Append the rank to the letter to form the row label:
  even rank $m$: $m=0 \to$ Table~4b, $m=2 \to$ Table~4d, $m=4 \to$ Table~4f;
  odd rank $n$: $n=1 \to$ Table~4c, $n=3 \to$ Table~4e.
\item Look up that row in the appropriate table for the independent components.
\end{enumerate}

\noindent\textbf{Example:} Point group $422$ has axial-even symbol $H_m$ (Table~4a).
For the rank-2 axial tensor, look up row $H_2$ in Table~4d; for rank-4, row $H_4$
in Table~4f.

\subsection*{Finding tensor forms for a magnetic point group}
\begin{enumerate}\setlength{\itemsep}{1pt}
\item Find the magnetic point group in \textbf{Table~6} to see its symmetry operations,
  classical subgroup, and generating matrices.
\item In \textbf{Table~7}, read the i-tensor and c-tensor symbol classes.
\item Use the same rank-mapping as above (Tables~4b--4f) to find the independent
  components.
\end{enumerate}

\noindent The 32 grey (type~II) groups are omitted from Tables~6 and~7: for each,
the i-tensor equals that of the classical group and every c-tensor vanishes.
